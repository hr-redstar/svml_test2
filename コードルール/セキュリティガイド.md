## セキュリティ基本原則

### 最小権限の原則
- **Discord Bot権限**: 必要最小限の権限のみ付与
- **Google Cloud権限**: IAMロールの最小化
- **ファイルアクセス**: 必要なファイルのみアクセス可能

### セキュアコーディング
```javascript
// ✅ 入力値検証
function validateInput(input) {
  if (!input || typeof input !== 'string') {
    throw new Error('無効な入力値');
  }
  
  // SQLインジェクション対策
  const sanitized = input.replace(/['"\\]/g, '');
  
  // 長さ制限
  if (sanitized.length > 255) {
    throw new Error('入力値が長すぎます');
  }
  
  return sanitized;
}

// ✅ 権限チェック
async function checkPermissions(interaction, requiredRole) {
  const member = interaction.member;
  
  if (!member.roles.cache.has(requiredRole)) {
    await interaction.reply({
      content: '❌ この操作を実行する権限がありません。',
      ephemeral: true
    });
    return false;
  }
  
  return true;
}
```

#### 必要な権限

### GCS ユーティリティ

#### 基本的なGCS操作
```javascript
// common/gcs/gcsUtils.js
const { Storage } = require('@google-cloud/storage');
const logger = require('@root/common/logger');

class GCSUtils {
  constructor() {
    this.storage = new Storage({
      keyFilename: process.env.GOOGLE_APPLICATION_CREDENTIALS,
      projectId: 'star-discord'
    });
    this.bucketName = 'svml-bot-data';
  }

  async uploadFile(localFilePath, destinationFileName) {
    try {
      const bucket = this.storage.bucket(this.bucketName);
      
      await bucket.upload(localFilePath, {
        destination: destinationFileName,
        metadata: {
          cacheControl: 'no-cache',
        },
      });
      
      logger.info(`ファイルアップロード完了: ${destinationFileName}`);
      return true;
    } catch (error) {
      logger.error('GCSアップロードエラー:', error);
      throw error;
    }
  }

  async downloadFile(fileName, destination) {
    try {
      const bucket = this.storage.bucket(this.bucketName);
      const file = bucket.file(fileName);
      
      await file.download({ destination });
      
      logger.info(`ファイルダウンロード完了: ${fileName}`);
      return true;
    } catch (error) {
      logger.error('GCSダウンロードエラー:', error);
      throw error;
    }
  }

  async fileExists(fileName) {
    try {
      const bucket = this.storage.bucket(this.bucketName);
      const file = bucket.file(fileName);
      const [exists] = await file.exists();
      
      return exists;
    } catch (error) {
      logger.error('GCSファイル存在チェックエラー:', error);
      return false;
    }
  }

  async deleteFile(fileName) {
    try {
      const bucket = this.storage.bucket(this.bucketName);
      const file = bucket.file(fileName);
      
      await file.delete();
      
      logger.info(`ファイル削除完了: ${fileName}`);
      return true;
    } catch (error) {
      logger.error('GCSファイル削除エラー:', error);
      throw error;
    }
  }
}

module.exports = new GCSUtils();
```

#### CSV ファイル処理
```javascript
// common/gcs/gcsCsvHelper.js
const gcsUtils = require('./gcsUtils');
const logger = require('@root/common/logger');
const fs = require('fs').promises;
const path = require('path');

class GCSCsvHelper {
  constructor() {
    this.tempDir = path.join(__dirname, '../../temp');
  }

  async readCsvFromGCS(fileName) {
    try {
      // 一時ディレクトリ作成
      await fs.mkdir(this.tempDir, { recursive: true });
      
      const tempFilePath = path.join(this.tempDir, fileName);
      
      // GCSからダウンロード
      await gcsUtils.downloadFile(fileName, tempFilePath);
      
      // ファイル読み取り
      const csvData = await fs.readFile(tempFilePath, 'utf8');
      
      // 一時ファイル削除
      await fs.unlink(tempFilePath);
      
      return this.parseCsv(csvData);
    } catch (error) {
      logger.error('CSV読み取りエラー:', error);
      throw error;
    }
  }

  async writeCsvToGCS(fileName, data) {
    try {
      // CSVデータ生成
      const csvData = this.generateCsv(data);
      
      // 一時ディレクトリ作成
      await fs.mkdir(this.tempDir, { recursive: true });
      
      const tempFilePath = path.join(this.tempDir, fileName);
      
      // ローカルファイル作成
      await fs.writeFile(tempFilePath, csvData, 'utf8');
      
      // GCSにアップロード
      await gcsUtils.uploadFile(tempFilePath, fileName);
      
      // 一時ファイル削除
      await fs.unlink(tempFilePath);
      
      logger.info(`CSV保存完了: ${fileName}`);
      return true;
    } catch (error) {
      logger.error('CSV保存エラー:', error);
      throw error;
    }
  }

  parseCsv(csvData) {
    const lines = csvData.trim().split('\n');
    const headers = lines[0].split(',').map(h => h.trim());
    
    return lines.slice(1).map(line => {
      const values = line.split(',').map(v => v.trim());
      const obj = {};
      
      headers.forEach((header, index) => {
        obj[header] = values[index] || '';
      });
      
      return obj;
    });
  }

  generateCsv(data) {
    if (!data || data.length === 0) {
      return '';
    }
    
    const headers = Object.keys(data[0]);
    const csvHeaders = headers.join(',');
    
    const csvRows = data.map(row => {
      return headers.map(header => {
        const value = row[header] || '';
        // カンマやダブルクォートを含む場合はエスケープ
        return value.toString().includes(',') ? `"${value}"` : value;
      }).join(',');
    });
    
    return [csvHeaders, ...csvRows].join('\n');
  }
}

module.exports = new GCSCsvHelper();
```

// ファイルパスの検証
function validateFilePath(filePath) {
  // ディレクトリトラバーサル攻撃の防止
  if (filePath.includes('..') || filePath.includes('/')) {
    throw new Error('無効なファイルパス');
  }
  
  // 許可された拡張子のみ
  const allowedExtensions = ['.csv', '.json', '.txt'];
  const ext = path.extname(filePath);
  
  if (!allowedExtensions.includes(ext)) {
    throw new Error('許可されていないファイル形式');
  }
  
  return true;
}

#### 暗号化
```javascript
const crypto = require('crypto');

class DataEncryption {
  constructor() {
    this.algorithm = 'aes-256-gcm';
    this.key = crypto.scryptSync(process.env.ENCRYPTION_KEY, 'salt', 32);
  }

  encrypt(text) {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipher(this.algorithm, this.key, iv);
    
    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return {
      encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex')
    };
  }

  decrypt(encryptedData) {
    const decipher = crypto.createDecipher(
      this.algorithm,
      this.key,
      Buffer.from(encryptedData.iv, 'hex')
    );
    
    decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'hex'));
    
    let decrypted = decipher.update(encryptedData.encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }
}
```

## インシデント対応

### インシデント分類

#### 緊急度レベル
- **Critical**: サービス停止、データ漏洩
- **High**: セキュリティ侵害、権限昇格
- **Medium**: 異常なアクセス、パフォーマンス低下
- **Low**: 軽微なエラー、警告

### 対応手順

#### セキュリティインシデント
```javascript
async function handleSecurityIncident(incident) {
  logger.error('セキュリティインシデント発生:', incident);
  
  // 1. 即座にアクセスを停止
  if (incident.severity === 'CRITICAL') {
    await emergencyShutdown();
  }
  
  // 2. インシデント記録
  await recordIncident(incident);
  
  // 3. 関係者に通知
  await notifySecurityTeam(incident);
  
  // 4. 証拠保全
  await preserveEvidence(incident);
}

async function emergencyShutdown() {
  logger.error('緊急シャットダウン実行');
  
  // Botを一時停止
  client.destroy();
  
  // 外部通知
  await notifyEmergency();
}
```
